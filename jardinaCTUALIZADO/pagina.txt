<script>
/* -------- SONIDO BURBUJA -------- */
function sonidoBurbuja(){
  const sound=document.getElementById("bubbleSound");
  sound.currentTime=0;
  sound.play();
}

/* sonido en todos los enlaces y botones */
document.querySelectorAll("a,button").forEach(el=>{
  el.addEventListener("click",sonidoBurbuja);
});

/* botón comprar */
function irComprar(){
  sonidoBurbuja();
  setTimeout(()=>{ window.location.href="comprar.html"; },300);
}

/* corazones animados */
function corazones(){
  for(let i=0;i<4;i++){
    const c=document.createElement("div");
    c.className="corazon";
    c.innerHTML="💗";
    c.style.left=Math.random()*window.innerWidth+"px";
    c.style.top=window.innerHeight+"px";
    document.body.appendChild(c);
    setTimeout(()=>c.remove(),2000);
  }
}

/* flores cayendo */
function petalos(){
  const p=document.createElement("div");
  p.className="petalo";
  p.innerHTML="🌸";
  p.style.left=Math.random()*window.innerWidth+"px";
  p.style.animationDuration=(Math.random()*5+5)+"s";
  document.body.appendChild(p);
  setTimeout(()=>p.remove(),10000);
}
setInterval(petalos,600);
